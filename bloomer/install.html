:root{ /* learn more about it here: https://developer.mozilla.org/en-US/docs/Web/CSS/Using_CSS_custom_properties */

--body-bg-color:#FFF; /*this changes the overall page's background color*/
--body-bg-image: url(''); /*add the body image here*/
--body-bg-position: center; /*add the body image positioning here. this can be two values (ex: top center) */
--body-bg-repeat: repeat; /* add the body image repeatability here. you can change it to repeat, repeat-x, repeat-y, etc. */
--body-bg-size: contain; /*this changes how you want the bg image to stretch out. */
--body-font: BlinkMacSystemFont,-apple-system,"Segoe UI",Roboto,Oxygen,Ubuntu,Cantarell,"Fira Sans","Droid Sans","Helvetica Neue",Helvetica,Arial,sans-serif ; /*change the font-family of the entire site here*/
--body-font-size: 16px; /*change the font size of the entire site here*/
--body-color: #4a4a4a; /*change the font color of the entire site here*/

--link:#3273dc; /*change the links color here*/
--link-hover: #363636;  /*change the links hover color here*/

--banner-bg-color: #00d1b2; /*-- you can change the color of the banner bg here (if using transparent images) */
--banner-image:  url('http://placehold.it/1600x350/00d1b2/FFFFFF/?text=Banner+Image'); /* replace the background here */
--banner-position: center; /*-- by default, the banner is centered. this is where you can change it --*/
--banner-repeat: no-repeat; /*-- by default, it's set not to repeat. you can change it to repeat, repeat-x, repeat-y, etc. --*/
--banner-size: cover; /* this stretches the banner out to cover the whole area */
--banner-height: 350px; /*-- you can change the height of the banner here */

--comic-nav: #4a4a4a; /*This changes the color of the comic navigation! */
--comic-nav-hover: #777; /*This is what it changes to when you hover over it! */

--footer-bg-color: #fafafa; /* this changes the footer's background color */
--footer-bg-image: url(''); /* changes the footer background image */
--footer-bg-position: top center; /* changes the positioning of the background image */
--footer-bg-repeat: repeat;
--footer-bg-size: contain;
--footer-link: #3273dc;
--footer-link-hover: #363636;
}
